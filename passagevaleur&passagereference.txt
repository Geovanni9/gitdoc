int a = 5;
int b = 6;


// passage par valeur
int add(int xx, int yy)
{
    return xx+yy;
}

print(add(a, b));

// passage par référence   * <=> contenu de & <=> adresse de
int add(int ref x, int ref y)
{
    return *x + *y;
}

print(add(&a, &b));
